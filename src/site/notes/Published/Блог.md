      const posts = await Promise.all(

  async function loadAndParseFile(url) {
    try {
      const response = await fetch(url);
      const data = await response.json();
      const content = atob(data.content); // –î–µ–∫–æ–¥–∏—Ä—É–µ–º base64
      return parseMetadata(content, data.name);
    } catch (error) {
      console.error('–û—à–∏–±–∫–∞ –ø—Ä–∏ –∑–∞–≥—Ä—É–∑–∫–µ —Ñ–∞–π–ª–∞:', error);
      return null;
    }
  }

  function parseMetadata(content, filename) {
    // –ò–∑–≤–ª–µ–∫–∞–µ–º YAML —Ñ—Ä–æ–Ω—Ç–º–∞—Ç—Ç–µ—Ä
    const match = content.match(/^---([\s\S]*?)---/);
    if (!match) return null;

    const meta = {};
    const lines = match.trim().split('\n');

    lines.forEach(line => {
      const [key, ...valueParts] = line.split(':');
      const value = valueParts.join(':').trim().replace(/^["']|["']$/g, '');
      meta[key.trim()] = value;
    });

    return {
      title: meta.title || filename.replace('.md', ''),
      date: meta.date || new Date().toISOString().split('T'),
      excerpt: meta.excerpt || '–ù–µ—Ç –æ–ø–∏—Å–∞–Ω–∏—è',
      category: meta.category || '–°—Ç–∞—Ç—å—è',
      emoji: meta.emoji || 'üìù',
      filename: filename.replace('.md', ''),
      published: meta['dg-publish'] === 'true'
    };
  }

  function renderCards(posts) {
    const container = document.getElementById('blog-container');
    
    if (posts.length === 0) {
      container.innerHTML = '<p style="grid-column: 1/-1; text-align: center; color: #999;">–ù–µ—Ç –æ–ø—É–±–ª–∏–∫–æ–≤–∞–Ω–Ω—ã—Ö —Å—Ç–∞—Ç–µ–π</p>';
      return;
    }

    container.innerHTML = posts.map(post => `
      <a href="../Blog/${post.filename}" class="blog-card">
        <div class="blog-card-image">${post.emoji}</div>
        <div class="blog-card-content">
          <h3 class="blog-card-title">${post.title}</h3>
          <p class="blog-card-excerpt">${post.excerpt}</p>
          <div class="blog-card-meta">
            <span>${new Date(post.date).toLocaleDateString('ru-RU', {
              year: 'numeric',
              month: 'long',
              day: 'numeric'
            })}</span>
            <span class="blog-card-tag">${post.category}</span>
          </div>
        </div>
      </a>
    `).join('');
  }
