  async function loadAndParseFile(url, filename) {
    try {
      console.log(`‚¨áÔ∏è –ó–∞–≥—Ä—É–∂–∞—é URL: ${url}`);
      
      const response = await fetch(url);

  function parseMetadata(content, filename) {
    try {
      if (!content || typeof content !== 'string') {
        console.warn('‚ö†Ô∏è –ö–æ–Ω—Ç–µ–Ω—Ç –Ω–µ —è–≤–ª—è–µ—Ç—Å—è —Å—Ç—Ä–æ–∫–æ–π');
        return null;
      }

      let meta = {};
      let contentStart = 0;

      // ‚úÖ –ü–û–ü–´–¢–ö–ê 1: JSON –ë–õ–û–ö
      const jsonLineMatch = content.match(/^\s*\{\s*"[^"]+"/);
      if (jsonLineMatch) {
        console.log('üîç –û–±–Ω–∞—Ä—É–∂–µ–Ω–æ –Ω–∞—á–∞–ª–æ JSON');
        
        // –ò—â–µ–º –∫–æ–Ω–µ—Ü JSON –æ–±—ä–µ–∫—Ç–∞
        let braceCount = 0;
        let jsonEnd = -1;
        let inString = false;
        let escapeNext = false;
        
        for (let i = 0; i < content.length; i++) {
          const char = content[i];
          const prevChar = i > 0 ? content[i-1] : '';
          
          if (escapeNext) {
            escapeNext = false;
            continue;
          }
          
          if (char === '\\') {
            escapeNext = true;
            continue;
          }
          
          if (char === '"' && prevChar !== '\\') {
            inString = !inString;
            continue;
          }
          
          if (!inString) {
            if (char === '{') braceCount++;
            if (char === '}') {
              braceCount--;
              if (braceCount === 0) {
                jsonEnd = i + 1;
                break;
              }
            }
          }
        }
        
        if (jsonEnd > 0) {
          const jsonStr = content.substring(0, jsonEnd);
          console.log('üì¶ JSON —Å—Ç—Ä–æ–∫–∞:', jsonStr);
          try {
            meta = JSON.parse(jsonStr);
            console.log('‚úÖ JSON —Ä–∞—Å–ø–∞—Ä—Å–µ–Ω:', meta);
            contentStart = jsonEnd;
          } catch (e) {
            console.warn('‚ö†Ô∏è –û—à–∏–±–∫–∞ –ø–∞—Ä—Å–∏–Ω–≥–∞ JSON:', e.message);
          }
        }
      }

      // ‚úÖ –ü–û–ü–´–¢–ö–ê 2: YAML –ë–õ–û–ö (–µ—Å–ª–∏ JSON –Ω–µ —Å—Ä–∞–±–æ—Ç–∞–ª)
      if (Object.keys(meta).length === 0) {
        console.log('üîç –ü—Ä–æ–±—É–µ–º YAML —Ñ–æ—Ä–º–∞—Ç');
        
        const yamlMatch = content.match(/^---\s*\n([\s\S]*?)\n---/);
        if (yamlMatch && yamlMatch) {
          console.log('üì¶ YAML –Ω–∞–π–¥–µ–Ω');
          
          const yamlContent = yamlMatch;
          const lines = yamlContent.split('\n');
          
          lines.forEach((line, idx) => {
            if (!line || !line.trim()) return;
            if (typeof line !== 'string') return;
            
            const colonIdx = line.indexOf(':');
            if (colonIdx === -1) return;
            
            const key = line.substring(0, colonIdx).trim();
            let value = line.substring(colonIdx + 1).trim();
            
            if ((value.startsWith('"') && value.endsWith('"')) || 
                (value.startsWith("'") && value.endsWith("'"))) {
              value = value.slice(1, -1);
            }
            
            if (value === 'true') meta[key] = true;
            else if (value === 'false') meta[key] = false;
            else meta[key] = value;
          });
          
          console.log('‚úÖ YAML —Ä–∞—Å–ø–∞—Ä—Å–µ–Ω:', meta);
          contentStart = yamlMatch.length;
        }
      }

      // ‚úÖ –ò–ó–í–õ–ï–ö–ê–ï–ú –ó–ù–ê–ß–ï–ù–ò–Ø
      let published = false;
      
      // –ü—Ä–æ–≤–µ—Ä—è–µ–º dg-publish
      if (meta['dg-publish'] === true || meta['dg-publish'] === 'true') {
        published = true;
      }
      
      // Fallback –¥–ª—è –¥—Ä—É–≥–∏—Ö —Ñ–æ—Ä–º–∞—Ç–æ–≤
      if (Object.keys(meta).length > 0) {
        console.log('üìå Meta –æ–±—ä–µ–∫—Ç:', meta);
        console.log('üìå dg-publish –∑–Ω–∞—á–µ–Ω–∏–µ:', meta['dg-publish']);
      }

      const title = meta.title || meta.Title || '–ë–µ–∑ –Ω–∞–∑–≤–∞–Ω–∏—è';
      const date = meta.date || meta.Date || new Date().toISOString().split('T');
      let excerpt = meta.excerpt || meta.description || meta.Description || '';
      
      // –ï—Å–ª–∏ excerpt –ø—É—Å—Ç, –±–µ—Ä—ë–º –ø–µ—Ä–≤—ã–µ 100 —Å–∏–º–≤–æ–ª–æ–≤ –∫–æ–Ω—Ç–µ–Ω—Ç–∞
      if (!excerpt && contentStart < content.length) {
        excerpt = content.substring(contentStart, contentStart + 100).replace(/\n/g, ' ').trim() + '...';
      }
      if (!excerpt) excerpt = '–ù–µ—Ç –æ–ø–∏—Å–∞–Ω–∏—è';

      let category = '–°—Ç–∞—Ç—å—è';
      if (Array.isArray(meta.tags) && meta.tags.length > 0) {
        category = meta.tags;
      } else if (typeof meta.tags === 'string') {
        category = meta.tags;
      } else if (meta.category) {
        category = meta.category;
      }

      const emoji = meta.emoji || meta.Emoji || 'üìù';

      const result = {
        title: String(title),
        date: String(date),
        excerpt: String(excerpt),
        category: String(category),
        emoji: String(emoji),
        published: published
      };

      console.log('‚úÖ –§–∏–Ω–∞–ª—å–Ω—ã–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç:', result);
      return result;
    } catch (error) {
      console.error('‚ùå –û—à–∏–±–∫–∞ –ø—Ä–∏ —Ä–∞–∑–±–æ—Ä–µ –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö:', error);
      return null;
    }
  }

  function renderCards(posts) {
    const container = document.getElementById('blog-container');
    
    if (!container) {
      console.error('‚ùå –ö–æ–Ω—Ç–µ–π–Ω–µ—Ä –Ω–µ –Ω–∞–π–¥–µ–Ω!');
      return;
    }

    if (!posts || !Array.isArray(posts) || posts.length === 0) {
      container.innerHTML = '<p style="grid-column: 1/-1; text-align: center; color: #999;">–ù–µ—Ç —Å—Ç–∞—Ç–µ–π</p>';
      return;
    }

    const html = posts.map((post) => {
      let dateStr = '–î–∞—Ç–∞ –Ω–µ–∏–∑–≤–µ—Å—Ç–Ω–∞';
      try {
        if (post.date) {
          const date = new Date(String(post.date));
          if (!isNaN(date.getTime())) {
            dateStr = date.toLocaleDateString('ru-RU', {
              year: 'numeric',
              month: 'long',
              day: 'numeric'
            });
          }
        }
      } catch (e) {
        console.warn('‚ö†Ô∏è –û—à–∏–±–∫–∞ –¥–∞—Ç—ã:', e);
      }

      const title = String(post.title || '').replace(/[<>]/g, '');
      const excerpt = String(post.excerpt || '').replace(/[<>]/g, '').substring(0, 150);
      const category = String(post.category || '').replace(/[<>]/g, '');
      const emoji = String(post.emoji || 'üìù').substring(0, 2);

      return `
        <div class="blog-card">
          <div class="blog-card-image">${emoji}</div>
          <div class="blog-card-content">
            <h3 class="blog-card-title">${title}</h3>
            <p class="blog-card-excerpt">${excerpt}</p>
            <div class="blog-card-meta">
              <span>${dateStr}</span>
              <span class="blog-card-tag">${category}</span>
            </div>
          </div>
        </div>
      `;
    }).join('');

    container.innerHTML = html;
    console.log('‚ú® –ö–∞—Ä—Ç–æ—á–∫–∏ –æ—Ç—Ä–µ–Ω–¥–µ—Ä–µ–Ω—ã!');
  }
