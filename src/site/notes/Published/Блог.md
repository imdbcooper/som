    try {
      const response = await fetch(url);
      if (!response.ok) {

  function parseMetadata(content) {
    try {
      // –ò–∑–≤–ª–µ–∫–∞–µ–º YAML —Ñ—Ä–æ–Ω—Ç–º–∞—Ç—Ç–µ—Ä
      const match = content.match(/^---([\s\S]*?)---/);
      if (!match) {
        console.warn('–§—Ä–æ–Ω—Ç–º–∞—Ç—Ç–µ—Ä –Ω–µ –Ω–∞–π–¥–µ–Ω');
        return null;
      }

      const meta = {};
      const yamlContent = match.trim();
      const lines = yamlContent.split('\n');

      lines.forEach(line => {
        // –ü—Ä–æ–ø—É—Å–∫–∞–µ–º –ø—É—Å—Ç—ã–µ —Å—Ç—Ä–æ–∫–∏
        if (!line || !line.trim()) return;
        
        // –ü—Ä–æ–≤–µ—Ä—è–µ–º —á—Ç–æ —ç—Ç–æ —Å—Ç—Ä–æ–∫–∞
        if (typeof line !== 'string') {
          console.warn('–°—Ç—Ä–æ–∫–∞ –Ω–µ —è–≤–ª—è–µ—Ç—Å—è string:', line);
          return;
        }

        const colonIndex = line.indexOf(':');
        if (colonIndex === -1) return;

        const key = line.substring(0, colonIndex).trim();
        let value = line.substring(colonIndex + 1).trim();

        // –£–¥–∞–ª—è–µ–º –∫–∞–≤—ã—á–∫–∏ –µ—Å–ª–∏ –µ—Å—Ç—å
        value = value.replace(/^["']|["']$/g, '');

        meta[key] = value;
      });

      // –ü—Ä–æ–≤–µ—Ä—è–µ–º –æ–±—è–∑–∞—Ç–µ–ª—å–Ω—ã–µ –ø–æ–ª—è
      const dg_publish = meta['dg-publish'] || 'false';
      const published = dg_publish === 'true' || dg_publish === true;

      return {
        title: meta.title || '–ë–µ–∑ –Ω–∞–∑–≤–∞–Ω–∏—è',
        date: meta.date || new Date().toISOString().split('T'),
        excerpt: meta.excerpt || meta.description || '–ù–µ—Ç –æ–ø–∏—Å–∞–Ω–∏—è',
        category: meta.category || meta.tags || '–°—Ç–∞—Ç—å—è',
        emoji: meta.emoji || 'üìù',
        published: published
      };
    } catch (error) {
      console.error('–û—à–∏–±–∫–∞ –ø—Ä–∏ —Ä–∞–∑–±–æ—Ä–µ –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö:', error);
      return null;
    }
  }
