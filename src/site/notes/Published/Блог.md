  async function loadAndParseFile(url) {
    try {
      const response = await fetch(url);
      if (!response.ok) {
        console.error(`–û—à–∏–±–∫–∞ –∑–∞–≥—Ä—É–∑–∫–∏ —Ñ–∞–π–ª–∞: ${response.status}`);
        return null;
      }
      
      const content = await response.text();
      const parsed = parseMetadata(content);
      console.log('–†–∞—Å–ø–∞—Ä—Å–µ–Ω —Ñ–∞–π–ª:', parsed);
      return parsed;
    } catch (error) {
      console.error('–û—à–∏–±–∫–∞ –ø—Ä–∏ –∑–∞–≥—Ä—É–∑–∫–µ —Ñ–∞–π–ª–∞:', error);
      return null;
    }
  }

  function parseMetadata(content) {
    try {
      if (!content || typeof content !== 'string') {
        console.warn('–ö–æ–Ω—Ç–µ–Ω—Ç –Ω–µ —è–≤–ª—è–µ—Ç—Å—è —Å—Ç—Ä–æ–∫–æ–π');
        return null;
      }

      // ‚úÖ –ü–†–û–ë–£–ï–ú JSON –§–û–†–ú–ê–¢ –°–ù–ê–ß–ê–õ–ê
      let meta = {};
      
      // –ü—Ä–æ–≤–µ—Ä—è–µ–º –Ω–∞ JSON: { ... } ---
      const jsonMatch = content.match(/^(\{[\s\S]*?\})\s*\n---/);
      if (jsonMatch) {
        console.log('–û–±–Ω–∞—Ä—É–∂–µ–Ω JSON —Ñ—Ä–æ–Ω—Ç–º–∞—Ç—Ç–µ—Ä');
        try {
          meta = JSON.parse(jsonMatch);
          console.log('JSON —Ä–∞—Å–ø–∞—Ä—Å–µ–Ω:', meta);
        } catch (e) {
          console.warn('–û—à–∏–±–∫–∞ –ø–∞—Ä—Å–∏–Ω–≥–∞ JSON:', e);
          meta = {};
        }
      } else {
        // ‚úÖ –ü–†–û–ë–£–ï–ú YAML –§–û–†–ú–ê–¢
        console.log('–ü—Ä–æ–±—É–µ–º YAML —Ñ–æ—Ä–º–∞—Ç');
        const yamlMatch = content.match(/^---\s*\n([\s\S]*?)\n---/);
        
        if (!yamlMatch || !yamlMatch) {
          console.warn('–§—Ä–æ–Ω—Ç–º–∞—Ç—Ç–µ—Ä –Ω–µ –Ω–∞–π–¥–µ–Ω');
          return null;
        }

        const yamlContent = yamlMatch;
        const lines = yamlContent.split('\n').filter(line => {
          if (typeof line !== 'string') return false;
          return line.trim().length > 0;
        });

        lines.forEach((line, index) => {
          try {
            if (!line || typeof line !== 'string') return;

            const trimmedLine = String(line).trim();
            if (!trimmedLine) return;

            const colonIndex = trimmedLine.indexOf(':');
            if (colonIndex === -1) return;

            const key = String(trimmedLine.substring(0, colonIndex)).trim();
            let value = String(trimmedLine.substring(colonIndex + 1)).trim();

            // –£–¥–∞–ª—è–µ–º –∫–∞–≤—ã—á–∫–∏
            if ((value.startsWith('"') && value.endsWith('"')) || 
                (value.startsWith("'") && value.endsWith("'"))) {
              value = value.slice(1, -1);
            }

            // –ö–æ–Ω–≤–µ—Ä—Ç–∏—Ä—É–µ–º —Ç–∏–ø—ã
            if (value === 'true') {
              meta[key] = true;
            } else if (value === 'false') {
              meta[key] = false;
            } else {
              meta[key] = value;
            }
          } catch (e) {
            console.warn(`–û—à–∏–±–∫–∞ –ø—Ä–∏ –ø–∞—Ä—Å–∏–Ω–≥–µ —Å—Ç—Ä–æ–∫–∏ ${index}:`, e);
          }
        });
      }

      // ‚úÖ –ò–ó–í–õ–ï–ö–ê–ï–ú –ó–ù–ê–ß–ï–ù–ò–Ø –° –ü–†–ê–í–ò–õ–¨–ù–´–ú–ò –ù–ê–ó–í–ê–ù–ò–Ø–ú–ò –ö–õ–Æ–ß–ï–ô
      
      // dg-publish –º–æ–∂–µ—Ç –±—ã—Ç—å –∫–∞–∫ "dg-publish" —Ç–∞–∫ –∏ –≤ –¥—Ä—É–≥–∏—Ö —Ñ–æ—Ä–º–∞—Ç–∞—Ö
      let published = false;
      const publishValue = meta['dg-publish'] || meta['dgPublish'] || false;
      if (publishValue === true || publishValue === 'true') {
        published = true;
      }

      // title –º–æ–∂–µ—Ç –±—ã—Ç—å —Ä–∞–∑–Ω—ã—Ö —Ñ–æ—Ä–º–∞—Ç–æ–≤
      const title = meta.title || meta.Title || '–ë–µ–∑ –Ω–∞–∑–≤–∞–Ω–∏—è';

      // date
      const date = meta.date || meta.Date || new Date().toISOString().split('T');

      // excerpt/description
      const excerpt = meta.excerpt || meta.description || meta.Description || '–ù–µ—Ç –æ–ø–∏—Å–∞–Ω–∏—è';

      // category/tags
      let category = meta.category || meta.Category || '–°—Ç–∞—Ç—å—è';
      if (Array.isArray(meta.tags) && meta.tags.length > 0) {
        category = meta.tags;
      } else if (typeof meta.tags === 'string') {
        category = meta.tags;
      }

      // emoji
      const emoji = meta.emoji || meta.Emoji || 'üìù';

      console.log('–§–∏–Ω–∞–ª—å–Ω—ã–µ –º–µ—Ç–∞–¥–∞–Ω–Ω—ã–µ:', {
        title, date, excerpt, category, emoji, published
      });

      return {
        title: String(title),
        date: String(date),
        excerpt: String(excerpt),
        category: String(category),
        emoji: String(emoji),
        published: published
      };
    } catch (error) {
      console.error('–û—à–∏–±–∫–∞ –ø—Ä–∏ —Ä–∞–∑–±–æ—Ä–µ –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö:', error);
      return null;
    }
  }

  function renderCards(posts) {
    const container = document.getElementById('blog-container');
    
    if (!container) {
      console.error('–ö–æ–Ω—Ç–µ–π–Ω–µ—Ä –Ω–µ –Ω–∞–π–¥–µ–Ω!');
      return;
    }

    if (!posts || !Array.isArray(posts) || posts.length === 0) {
      container.innerHTML = '<p style="grid-column: 1/-1; text-align: center; color: #999;">–ù–µ—Ç —Å—Ç–∞—Ç–µ–π –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è</p>';
      return;
    }

    const html = posts.map((post) => {
      // –ë–µ–∑–æ–ø–∞—Å–Ω–æ —Ñ–æ—Ä–º–∞—Ç–∏—Ä—É–µ–º –¥–∞—Ç—É
      let dateStr = '–î–∞—Ç–∞ –Ω–µ–∏–∑–≤–µ—Å—Ç–Ω–∞';
      try {
        if (post.date) {
          const date = new Date(String(post.date));
          if (!isNaN(date.getTime())) {
            dateStr = date.toLocaleDateString('ru-RU', {
              year: 'numeric',
              month: 'long',
              day: 'numeric'
            });
          }
        }
      } catch (e) {
        console.warn('–û—à–∏–±–∫–∞ —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—è –¥–∞—Ç—ã:', e);
      }

      // –ë–µ–∑–æ–ø–∞—Å–Ω–æ —ç–∫—Ä–∞–Ω–∏—Ä—É–µ–º HTML
      const title = String(post.title || '').replace(/[<>]/g, '');
      const excerpt = String(post.excerpt || '').replace(/[<>]/g, '');
      const category = String(post.category || '').replace(/[<>]/g, '');
      const emoji = String(post.emoji || 'üìù').substring(0, 2);

      return `
        <div class="blog-card">
          <div class="blog-card-image">${emoji}</div>
          <div class="blog-card-content">
            <h3 class="blog-card-title">${title}</h3>
            <p class="blog-card-excerpt">${excerpt}</p>
            <div class="blog-card-meta">
              <span>${dateStr}</span>
              <span class="blog-card-tag">${category}</span>
            </div>
          </div>
        </div>
      `;
    }).join('');

    container.innerHTML = html;
  }
